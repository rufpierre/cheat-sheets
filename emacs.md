save : `C-x C-s`

quit : `C-x C-c`

beginning of line : `C-a`

end of line : `C-e`

beginning of file : `ESC <`

end of file : `ESC >`

delete from cursor : `C-k`

begin selection : `C-SPACE`

cut selection : `C-w`

paste : `C-y`

***stop any command*** : `C-g`

undo : `C-x u`

open file : `C-x C-f`

display buffer list : `C-x C-b`

open buffer : `C-x b`
